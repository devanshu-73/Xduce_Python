UPDATE pcc
SET Technical_adequacy = 
    CASE 
        WHEN cco.TECH_ANNUAL_PREMIUM = 0 THEN 0
        ELSE (cco.CHRG_ANNUAL_PREM / cco.TECH_ANNUAL_PREMIUM) * 100
    END
FROM pric.cybercoveragecommon pcc
INNER JOIN cybercoverageoption cco
ON pcc.CoverageID = cco.CoverageID;
